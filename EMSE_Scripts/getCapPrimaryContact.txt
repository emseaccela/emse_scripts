capIDScriptResult = aa.cap.getCapIDByIDValue(spc, capIDValue, auditID)

if( capIDScriptResult.getSuccess())
{
	capIDScriptModel = capIDScriptResult.getOutput();
	result = aa.cap.getCapPrimaryContact( capIDScriptModel);

	}
	else
	{

	}
}
else
{
	aa.print("Failed:" + capIDScriptResult.getErrorMessage());
}
