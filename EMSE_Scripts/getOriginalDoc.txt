//DocumentModel
var result;
var resubmitDocModel;
var documentID = "10000087";

resubmitDocModelResult = aa.document.getDocumentByPK( documentID );
if(resubmitDocModelResult.getSuccess())
{
	resubmitDocModel = resubmitDocModelResult.getOutput();
	result = aa.document.getOriginalDoc(resubmitDocModel);
	if (result.getSuccess())
	{
	    aa.print("Success");
	}
	else
	{
	    aa.print("Failed:" + result.getErrorMessage());
	}
}
else
{
	aa.print("Failed:" + resubmitDocModelResult.getErrorMessage());
}
