pCapContactNumber = 50331230;
var capContactScriptArray;
var result;
var contactList = new java.util.ArrayList();

capContactScriptArrayResult = aa.people.getCapContactByContactID( pCapContactNumber )
if(capContactScriptArrayResult.getSuccess())
{
	capContactScriptArray = capContactScriptArrayResult.getOutput();
	
	for(i=0; i < capContactScriptArray.length; i++)
	{
		capContactModel = capContactScriptArray[i].getCapContactModel();
		//edit
		capContactModel.setCity("SZ");
		contactList.add( capContactModel );
	}
	
	result = aa.people.newContactByCapContacts(contactList);
	if (result.getSuccess())
	{
	    aa.print("Success");
	}
	else
	{
	    aa.print("Failed:" + result.getErrorMessage());
	}

}
else
{
	aa.print("Failed:" + capContactScriptArrayResult.getErrorMessage());
}
